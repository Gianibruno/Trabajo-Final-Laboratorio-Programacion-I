#OBTENER TODOS LOS LECTORES MOROSOS:
SELECT DISTINCT l.* 
FROM lectores AS l, prestamos AS p
WHERE p.id_lector = l.id_lector
AND p.fecha_prestamo + INTERVAL 30 DAY < NOW()
AND p.fecha_devolucion IS NULL;

#OBTENER LECTORES CON MULTA:
SELECT l.* 
FROM lectores AS l, prestamos AS p, multa AS m
WHERE p.id_lector = l.id_lector
AND p.id_prestamo = m.id_prestamo;
